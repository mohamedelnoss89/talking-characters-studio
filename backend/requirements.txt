# Talking Characters Studio — Backend Python dependencies
#
# This file is the SINGLE source of truth for the Python backend's
# third-party packages. It is consumed by:
#   - developers running the backend manually: `pip install -r requirements.txt`
#   - the Electron installer (desktop/src/installer-python.js reads this
#     file at install time and feeds it to `pip install -r`)
#   - CI/CD pipelines that need to reproduce the Python environment
#
# Do NOT duplicate this list in installer-python.js or README.md — keep
# them in sync by having them read this file.
#
# Pinned versions match what the Electron installer previously hard-coded,
# to avoid changing runtime behaviour. Newer versions can be bumped here
# and tested in isolation.

# --- Deep learning stack ---
# torch 2.2.x is the last version that ships pre-built CPU wheels for
# Python 3.10/3.11 on Windows without needing a CUDA toolkit install.
torch==2.2.2
torchvision==0.17.2
torchaudio==2.2.2

# --- Computer vision + face AI ---
opencv-python==4.9.0.80
mediapipe==0.10.14

# GFPGAN pulls in basicsr as a transitive dependency, but we pin basicsr
# explicitly because newer basicsr versions break GFPGAN's API on import.
gfpgan==1.3.8
basicsr==1.4.2

# --- Audio / signal processing ---
librosa==0.10.1
# scipy is a transitive dep of librosa + mediapipe, but hand_gesture.py
# imports it directly (scipy.signal.find_peaks), so we list it explicitly
# to make the dependency clear and avoid silent failures if librosa ever
# drops it.
scipy>=1.10,<2

# --- Numerics ---
# numpy<2 is required because torch 2.2.x and mediapipe 0.10.14 were
# built against numpy 1.x ABI. numpy>=2 causes "module compiled against
# API version 0x10 but this version of numpy is 0x11" runtime errors.
numpy<2
Pillow>=9,<12

# --- Web framework ---
fastapi==0.109.2
uvicorn==0.27.1
python-multipart==0.0.9
pydantic>=2,<3

# --- TTS ---
edge-tts>=6.1,<7

# --- Utilities ---
tqdm>=4.65,<5
requests>=2.28,<3
huggingface_hub>=0.20,<1
